using Microsoft.EntityFrameworkCore;
using Zek.Data.Entity;

namespace Zek.Model.Membership.Identity
{
    public class UserRole : Microsoft.AspNetCore.Identity.EntityFrameworkCore.IdentityUserRole<int> { }

    public class UserRoleMap : EntityTypeMap<UserRole>
    {
        public UserRoleMap(ModelBuilder builder) : base(builder)
        {
            ToTable("T_UserRole", "Membership");
        }
    }
}
