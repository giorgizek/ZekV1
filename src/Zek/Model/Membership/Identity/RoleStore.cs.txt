using System;
using System.Security.Claims;
using Microsoft.EntityFrameworkCore;

namespace Zek.Model.Membership.Identity
{
    //public class RoleStore : Microsoft.AspNetCore.Identity.EntityFrameworkCore.RoleStore<Role, DbContext, int, UserRole, RoleClaim>
    public class RoleStore : Microsoft.AspNetCore.Identity.EntityFrameworkCore.RoleStore<Role, DbContext, int, UserRole, Microsoft.AspNetCore.Identity.EntityFrameworkCore.IdentityRoleClaim<int>>
    {
        public RoleStore(DbContext context, Microsoft.AspNetCore.Identity.IdentityErrorDescriber describer = null) : base(context, describer)
        {
        }

        protected override Microsoft.AspNetCore.Identity.EntityFrameworkCore.IdentityRoleClaim<int> CreateRoleClaim(Role role, Claim claim) => new RoleClaim
        {
            RoleId = role.Id,
            ClaimType = claim.Type,
            ClaimValue = claim.Value
        };

        //protected override RoleClaim CreateRoleClaim(Role role, Claim claim) => new RoleClaim

        //    ClaimValue = claim.Value
        //};
    }
}
