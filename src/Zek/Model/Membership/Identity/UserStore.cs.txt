using System;
using System.Security.Claims;
using Microsoft.EntityFrameworkCore;

namespace Zek.Model.Membership.Identity
{
    public class UserStore : Microsoft.AspNetCore.Identity.EntityFrameworkCore.UserStore<User, Role, DbContext, int, UserClaim, UserRole, UserLogin, UserToken>
    {
        public UserStore(DbContext context, Microsoft.AspNetCore.Identity.IdentityErrorDescriber describer = null) : base(context, describer)
        {
        }

        protected override UserClaim CreateUserClaim(User user, Claim claim)
        {
            var userClaim = new UserClaim
            {
                UserId = user.Id
            };
            userClaim.InitializeFromClaim(claim);
            return userClaim;
        }

        protected override UserLogin CreateUserLogin(User user, Microsoft.AspNetCore.Identity.UserLoginInfo login) => new UserLogin
        {
            UserId = user.Id,
            ProviderKey = login.ProviderKey,
            LoginProvider = login.LoginProvider,
            ProviderDisplayName = login.ProviderDisplayName
        };

        protected override UserRole CreateUserRole(User user, Role role) => new UserRole
        {
            UserId = user.Id,
            RoleId = role.Id
        };

        protected override UserToken CreateUserToken(User user, string loginProvider, string name, string value) => new UserToken
        {
            UserId = user.Id,
            LoginProvider = loginProvider,
            Name = name,
            Value = value
        };
    }
}
