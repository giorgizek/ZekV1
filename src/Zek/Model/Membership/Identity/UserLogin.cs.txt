using Microsoft.EntityFrameworkCore;
using Zek.Data.Entity;

namespace Zek.Model.Membership.Identity
{
    public class UserLogin : Microsoft.AspNetCore.Identity.EntityFrameworkCore.IdentityUserLogin<int> { }

    public class UserLoginMap : EntityTypeMap<UserLogin>
    {
        public UserLoginMap(ModelBuilder builder) : base(builder)
        {
            ToTable("T_UserLogin", "Membership");
        }
    }
}
